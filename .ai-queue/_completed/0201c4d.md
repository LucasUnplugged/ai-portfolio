# Segment 1: shadcn/ui Init, Base Components & Font Setup

## Objective
Initialize shadcn/ui with Tailwind CSS 4, install base components, set up all Google Fonts in the root layout, and configure the main site's globals.css with a neutral shadcn theme.

## Scope
- `components.json` (new — created by shadcn init)
- `src/lib/utils.ts` (new — created by shadcn init)
- `src/app/globals.css` (modify — replace with shadcn neutral theme)
- `src/app/layout.tsx` (modify — add all Google Fonts)
- `src/components/ui/**` (new — shadcn components)
- `package.json` (modify — new deps)
- `package-lock.json` (modify — lockfile update)

## Instructions

### Step 1: Initialize shadcn/ui

Run shadcn init for Next.js with Tailwind CSS 4:

```bash
npx shadcn@latest init -d
```

If interactive prompts appear, choose:
- Style: default (New York is also fine if default isn't available)
- Base color: neutral
- CSS variables: yes

This should create `components.json` and `src/lib/utils.ts`, and update `globals.css`.

### Step 2: Install base shadcn components

```bash
npx shadcn@latest add button card badge separator avatar dropdown-menu tabs
```

### Step 3: Install lucide-react

```bash
npm install lucide-react
```

1. Import tailwindcss
2. Define CSS custom properties for light mode under `:root`
3. Define dark mode overrides under `.dark`

### Step 5: Configure all Google Fonts in root layout

Update `src/app/layout.tsx` to load ALL needed fonts via `next/font/google`. Each font gets a CSS variable.

```tsx
import type { Metadata } from "next";
import {
  Geist,
  Geist_Mono,
} from "next/font/google";
import {
  Manrope,
  Bricolage_Grotesque,
  ADLaM_Display,
  Afacad,
  Adamina,
  Fira_Code,
  Inter,
  JetBrains_Mono,
} from "next/font/google";
import "./globals.css";

// Main site fonts
const geistSans = Geist({
  variable: "--font-geist-sans",
  subsets: ["latin"],
});

const geistMono = Geist_Mono({
  variable: "--font-geist-mono",
  subsets: ["latin"],
});

// Ledger demo fonts
const manrope = Manrope({
  variable: "--font-manrope",
  subsets: ["latin"],
});

const bricolageGrotesque = Bricolage_Grotesque({
  variable: "--font-bricolage-grotesque",
  subsets: ["latin"],
});

const adlamDisplay = ADLaM_Display({
  variable: "--font-adlam-display",
  subsets: ["latin"],
  weight: "400",
});

// Ritual demo fonts
const afacad = Afacad({
  variable: "--font-afacad",
  subsets: ["latin"],
});

const adamina = Adamina({
  variable: "--font-adamina",
  subsets: ["latin"],
  weight: "400",
});

const firaCode = Fira_Code({
  variable: "--font-fira-code",
  subsets: ["latin"],
});

// Circles demo fonts
const inter = Inter({
  variable: "--font-inter",
  subsets: ["latin"],
});

const jetbrainsMono = JetBrains_Mono({
  variable: "--font-jetbrains-mono",
  subsets: ["latin"],
});

export const metadata: Metadata = {
  title: "Lucas Castro — Product Engineer",
  description:
    "Portfolio showcasing UI design and frontend engineering through interactive case studies.",
};

export default function RootLayout({
  children,
}: Readonly<{
  children: React.ReactNode;
}>) {
  return (
    <html lang="en">
      <body
        className={`${geistSans.variable} ${geistMono.variable} ${manrope.variable} ${bricolageGrotesque.variable} ${adlamDisplay.variable} ${afacad.variable} ${adamina.variable} ${firaCode.variable} ${inter.variable} ${jetbrainsMono.variable} antialiased`}
      >
        {children}
      </body>
    </html>
  );
}
```

Note: ADLaM_Display and Adamina are single-weight fonts (400 only), so they need an explicit `weight: "400"` parameter.
