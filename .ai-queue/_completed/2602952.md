### Step 5: Create demo layout files

Each demo layout wraps children in a theme-scoped `<div>` and imports its CSS file.

**`src/app/app/ledger/layout.tsx`:**
```tsx
import type { ReactNode } from "react";
import "./globals.css";

export default function LedgerLayout({ children }: { children: ReactNode }) {
  return (
    <div className="ledger-theme min-h-screen bg-background text-foreground">
      {children}
    </div>
  );
}
```
